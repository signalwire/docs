import APIField from "@site/src/components/APIField";

[swaig.functions]: /swml/methods/ai/swaig/functions

export default function StepTable() {
  return (
    <>
      <APIField
        name="name"
        type="string"
        required={true}
      >
        The name of the step. The name must be unique within the context. The name is used for referencing the step in the context.
      </APIField>

      <APIField
        name="end"
        type="boolean"
        default="`false`"
      >
        A boolean value that determines if the step is the last in the context. If `true`, the context ends after this step.<br />**Important**: This **cannot** be used along with the `valid_steps` parameter.
      </APIField>

      <APIField
        name="functions"
        type="string[]"
      >
        An array of strings, where each string is the name of a [SWAIG.function][swaig.functions] that can be executed from this step.
      </APIField>

      <APIField
        name="step_criteria"
        type="string"
      >
        The criteria that must be met for the AI to proceed to the next step. The criteria is a instruction given to the AI. It's **highly** recommended you create a custom criteria for the step to get the intended behavior.
      </APIField>

      <APIField
        name="skip_user_turn"
        type="boolean"
        default="`false`"
      >
        A boolean value, if set to `true`, will skip the user's turn to respond in the conversation and proceed to the next step.
      </APIField>

      <APIField
        name="valid_contexts"
        type="string[]"
      >
        An array of context names that the AI can transition to from this step. This must be a valid `contexts.name` that is present in your [`contexts`](../index.mdx) object.
      </APIField>

      <APIField
        name="valid_steps"
        type="string[]"
        default="Proceeds to the next step."
      >
        An array of valid step names that the conversation can proceed to from this step. If the array is empty, or the `valid_steps` key is not present, the conversation will proceed to the next step in the context.<br />**Important**: This **cannot** be used along with the `end` parameter.
      </APIField>
    </>
  );
}
