import APIField from "@site/src/components/APIField";

<APIField
  name="play.auto_answer"
  type="boolean"
  default="true"
>
  If `true`, the call will automatically answer as the sound is playing. If `false`, you will start playing the audio during early media, additionally, you can instruct to answer the call with the [`answer method`](/swml/methods/answer).
</APIField>

<APIField
  name="play.volume"
  type="number"
  default="0"
>
  Volume gain to apply to played URLs. Allowed values from `-40.0` to `40.0`.
</APIField>

<APIField
  name="play.say_voice"
  type="string"
  default="Polly.Salli"
>
  Voice to use with `say:` for text to speech.
</APIField>

<APIField
  name="play.say_language"
  type="string"
  default="en-US"
>
  Language to use with `say:` for text to speech.
</APIField>

<APIField
  name="play.say_gender"
  type="string"
  default="female"
>
  Gender to use with `say:` for text to speech.
</APIField>

<APIField
  name="play.status_url"
  type="string"
>
  HTTP or HTTPS URL to deliver play status events.
</APIField>
