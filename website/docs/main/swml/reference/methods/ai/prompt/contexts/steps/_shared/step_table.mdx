import APIField from "@site/src/components/APIField";

[swaig.functions]: /swml/methods/ai/swaig/functions

<APIField
  name="steps[].name"
  type="string"
  required={true}
>
  The name of the step. The name must be unique within the context. The name is used for referencing the step in the context.
</APIField>

<APIField
  name="steps[].end"
  type="boolean"
  default="false"
>
  A boolean value that determines if the step is the last in the context. If <code>true</code>, the context ends after this step.

This parameter is required if you want to end a step in a context itself. <br /><strong>Important</strong>: This <strong>cannot</strong> be used along with the <code>valid_steps</code> parameter.
</APIField>

<APIField
  name="steps[].functions"
  type="string[]"
>
  An array of strings, where each string is the name of a <a href="/swml/methods/ai/swaig/functions">SWAIG.function</a> that can be executed from this step.
</APIField>

<APIField
  name="steps[].step_criteria"
  type="string"
>
  The criteria that must be met for the AI to proceed to the next step. The criteria is a instruction given to the AI. It's <strong>highly</strong> recommended you create a custom criteria for the step to get the intended behavior.
</APIField>

<APIField
  name="steps[].skip_user_turn"
  type="boolean"
  default="false"
>
  A boolean value, if set to <code>true</code>, will skip the user's turn to respond in the conversation and proceed to the next step.
</APIField>

<APIField
  name="steps[].valid_contexts"
  type="string[]"
>
  An array of context names that the AI can transition to from this step. This must be a valid <code>contexts.name</code> that is present in your <a href="/swml/methods/ai/prompt/contexts"><code>contexts</code></a> object.
</APIField>

<APIField
  name="steps[].valid_steps"
  type="string[]"
  default="Proceeds to the next step."
>
  An array of valid step names that the conversation can proceed to from this step. If the array is empty, or the <code>valid_steps</code> key is not present, the conversation will proceed to the next step in the context.<br /><strong>Important</strong>: This <strong>cannot</strong> be used along with the <code>end</code> parameter.
</APIField>
