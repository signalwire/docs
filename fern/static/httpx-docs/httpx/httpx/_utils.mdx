# Wildcard matching...
&gt;&gt;&gt; pattern = URLPattern("all://")
&gt;&gt;&gt; pattern.matches(httpx.URL("http://example.com"))
True

# Witch scheme matching...
&gt;&gt;&gt; pattern = URLPattern("https://")
&gt;&gt;&gt; pattern.matches(httpx.URL("https://example.com"))
True
&gt;&gt;&gt; pattern.matches(httpx.URL("http://example.com"))
False

# With domain matching...
&gt;&gt;&gt; pattern = URLPattern("https://example.com")
&gt;&gt;&gt; pattern.matches(httpx.URL("https://example.com"))
True
&gt;&gt;&gt; pattern.matches(httpx.URL("http://example.com"))
False
&gt;&gt;&gt; pattern.matches(httpx.URL("https://other.com"))
False

# Wildcard scheme, with domain matching...
&gt;&gt;&gt; pattern = URLPattern("all://example.com")
&gt;&gt;&gt; pattern.matches(httpx.URL("https://example.com"))
True
&gt;&gt;&gt; pattern.matches(httpx.URL("http://example.com"))
True
&gt;&gt;&gt; pattern.matches(httpx.URL("https://other.com"))
False

# With port matching...
&gt;&gt;&gt; pattern = URLPattern("https://example.com:1234")
&gt;&gt;&gt; pattern.matches(httpx.URL("https://example.com:1234"))
True
&gt;&gt;&gt; pattern.matches(httpx.URL("https://example.com"))
False
