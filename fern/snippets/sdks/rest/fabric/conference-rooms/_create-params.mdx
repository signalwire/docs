<ParamField path="name" type="string" required={true} toc={true}>
  The name of the conference room.
</ParamField>

<ParamField path="enable_room_previews" type="boolean" required={true} toc={true}>
  Enables live video room previews for the conference.
</ParamField>

<ParamField path="display_name" type="string" toc={true}>
  Display name of the conference room. Maximum 200 characters.
</ParamField>

<ParamField path="description" type="string" toc={true}>
  The description of the conference room. Maximum 3000 characters.
</ParamField>

<ParamField path="join_from" type="string" toc={true}>
  The time users are allowed to start joining the conference. Joining before this time will result
  in failure to join the conference. ISO 8601 date-time format.
</ParamField>

<ParamField path="join_until" type="string" toc={true}>
  The time until which users are allowed to join the conference. Attempting to join after this time
  will result in failure. ISO 8601 date-time format.
</ParamField>

<ParamField path="max_members" type="integer" toc={true}>
  Maximum number of members allowed in the conference room. Range: `0` to `300`.
</ParamField>

<ParamField path="quality" type="string" default="720p" toc={true}>
  The video quality of the conference room. Valid values: `"1080p"`, `"720p"`.
</ParamField>

<ParamField path="remove_at" type="string" toc={true}>
  The time to remove all participants from the conference. ISO 8601 date-time format.
</ParamField>

<ParamField path="remove_after_seconds_elapsed" type="integer" toc={true}>
  The amount of time in seconds to remove a participant from a conference after they join. Range: `0` to `200000`.
</ParamField>

<ParamField path="layout" type="string" default="grid-responsive" toc={true}>
  The video layout of the conference.
</ParamField>

<ParamField path="record_on_start" type="boolean" toc={true}>
  Starts recording when the conference starts.
</ParamField>

<ParamField path="meta" type="object" toc={true}>
  Metadata of the conference. A key-value object.
</ParamField>

<ParamField path="sync_audio_video" type="boolean" toc={true}>
  Syncs the participants' audio and video.
</ParamField>

<ParamField path="tone_on_entry_and_exit" type="boolean" toc={true}>
  Plays a tone when a participant joins or leaves the conference.
</ParamField>

<ParamField path="room_join_video_off" type="boolean" toc={true}>
  Turns the conference video off when the participant joins the room if `true`.
</ParamField>

<ParamField path="user_join_video_off" type="boolean" toc={true}>
  Turns the participant's video off when the participant joins the room if `true`.
</ParamField>
