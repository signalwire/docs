<ParamField path="To" type="string" required={true} toc={true}>
  The destination phone number in E.164 format.
</ParamField>

<ParamField path="From" type="string" toc={true}>
  The sender phone number (must be a number you own). Either `From` or
  `MessagingServiceSid` must be provided.
</ParamField>

<ParamField path="Body" type="string" toc={true}>
  The text body of the message. Required for SMS; optional for MMS if `MediaUrl` is provided.
</ParamField>

<ParamField path="MediaUrl" type="string | array" toc={true}>
  URL or array of URLs of media to include (for MMS). Up to 10 media URLs.
</ParamField>

<ParamField path="StatusCallback" type="string" toc={true}>
  URL to receive message status webhook events.
</ParamField>

<ParamField path="MessagingServiceSid" type="string" toc={true}>
  The SID of the messaging service to use. Either `From` or `MessagingServiceSid` must be provided.
</ParamField>

<ParamField path="ApplicationSid" type="string" toc={true}>
  The SID of the application to handle status callbacks.
</ParamField>

<ParamField path="MaxPrice" type="string" toc={true}>
  Maximum price in USD to pay for the message.
</ParamField>

<ParamField path="ValidityPeriod" type="number" default="14400" toc={true}>
  Time in seconds that the message is valid. After this period, delivery will not be attempted.
</ParamField>

<ParamField path="SendAsMms" type="boolean" toc={true}>
  Whether to send as MMS even if the body is text-only.
</ParamField>
