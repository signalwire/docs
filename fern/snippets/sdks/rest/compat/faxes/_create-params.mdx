<ParamField path="To" type="string" required={true} toc={true}>
  The destination fax number in E.164 format.
</ParamField>

<ParamField path="From" type="string" required={true} toc={true}>
  The sender number (must be a number you own).
</ParamField>

<ParamField path="MediaUrl" type="string" required={true} toc={true}>
  URL of the document to fax (PDF format).
</ParamField>

<ParamField path="StatusCallback" type="string" toc={true}>
  URL to receive fax status webhook events.
</ParamField>

<ParamField path="StatusCallbackMethod" type="string" toc={true}>
  HTTP method for the `StatusCallback`. `"GET"` or `"POST"`.
</ParamField>

<ParamField path="StatusCallbackEvent" type="array" toc={true}>
  Fax status events to receive at the callback.
</ParamField>

<ParamField path="Quality" type="string" toc={true}>
  Fax quality setting.

  - `"standard"` -- standard quality
  - `"fine"` -- fine quality
  - `"superfine"` -- highest quality
</ParamField>

<ParamField path="StoreMedia" type="string" toc={true}>
  Whether to store the fax media after sending. `"true"` or `"false"`.
</ParamField>

<ParamField path="Ttl" type="number" toc={true}>
  Time to live in minutes. The fax will be canceled if not completed within this time.
</ParamField>

<ParamField path="SipAuthUsername" type="string" toc={true}>
  SIP authentication username for SIP fax destinations.
</ParamField>

<ParamField path="SipAuthPassword" type="string" toc={true}>
  SIP authentication password for SIP fax destinations.
</ParamField>
