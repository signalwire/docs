<ParamField path="number" type="string" required={true} toc={true}>
  The phone number to import in E.164 format (e.g., `"+15559876543"`).
</ParamField>

<ParamField path="number_type" type="string" required={true} toc={true}>
  The type of phone number being imported (e.g., `"longcode"`).
</ParamField>

<ParamField path="capabilities" type="array" toc={true}>
  Capabilities to enable for this number. Defaults to all (`sms`, `voice`, `fax`, `mms`).
  Must include at least one if provided.
</ParamField>
