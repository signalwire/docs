<ParamField path="Muted" type="boolean" toc={true}>
  Set to `true` to mute or `false` to unmute.
</ParamField>

<ParamField path="Hold" type="boolean" toc={true}>
  Set to `true` to hold or `false` to unhold.
</ParamField>

<ParamField path="AnnounceUrl" type="string" toc={true}>
  URL returning LAML to play to the participant.
</ParamField>

<ParamField path="AnnounceMethod" type="string" default="POST" toc={true}>
  HTTP method to use when requesting the `AnnounceUrl`. `"GET"` or `"POST"`.
</ParamField>

<ParamField path="Coaching" type="boolean" toc={true}>
  Set to `true` to enable coaching mode for this participant.
</ParamField>

<ParamField path="CallSidToCoach" type="string" toc={true}>
  The call SID of the participant to coach.
</ParamField>

<ParamField path="HoldUrl" type="string" toc={true}>
  URL returning LAML to play to the participant while on hold.
</ParamField>

<ParamField path="HoldMethod" type="string" default="GET" toc={true}>
  HTTP method to use when requesting the `HoldUrl`. `"GET"` or `"POST"`.
</ParamField>

<ParamField path="WaitUrl" type="string" toc={true}>
  URL returning LAML for hold music or messages while waiting.
</ParamField>

<ParamField path="WaitMethod" type="string" default="POST" toc={true}>
  HTTP method to use when requesting the `WaitUrl`. `"GET"` or `"POST"`.
</ParamField>
