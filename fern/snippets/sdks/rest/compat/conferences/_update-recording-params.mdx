<ParamField path="Status" type="string" required={true} toc={true}>
  New recording status.

  - `"paused"` -- pause the recording
  - `"in-progress"` -- resume a paused recording
  - `"stopped"` -- stop the recording permanently
</ParamField>

<ParamField path="PauseBehavior" type="string" default="silence" toc={true}>
  Behavior when pausing. `"skip"` discards audio while paused; `"silence"` inserts
  silence into the recording.
</ParamField>
