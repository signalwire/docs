<ParamField path="Url" type="string" required={true} toc={true}>
  The WebSocket URL to stream audio to.
</ParamField>

<ParamField path="Name" type="string" toc={true}>
  A friendly name for the stream.
</ParamField>

<ParamField path="Track" type="string" default="both_tracks" toc={true}>
  Which audio track to stream.

  - `"inbound_track"` — audio from the inbound leg
  - `"outbound_track"` — audio from the outbound leg
  - `"both_tracks"` — audio from both legs
</ParamField>

<ParamField path="StatusCallback" type="string" toc={true}>
  URL to receive stream status callbacks.
</ParamField>

<ParamField path="StatusCallbackMethod" type="string" default="POST" toc={true}>
  HTTP method for the `StatusCallback` URL. `"GET"` or `"POST"`.
</ParamField>

<ParamField path="Parameter1.Name" type="string" toc={true}>
  Name of a custom parameter to send to the WebSocket server. Up to 99 parameters supported (`Parameter1` through `Parameter99`).
</ParamField>

<ParamField path="Parameter1.Value" type="string" toc={true}>
  Value of the custom parameter specified by `Parameter1.Name`.
</ParamField>

<ParamField path="AuthorizationBearerToken" type="string" toc={true}>
  Bearer token for authenticating with the WebSocket server.
</ParamField>
