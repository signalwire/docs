<ParamField path="To" type="string" required={true} toc={true}>
  The destination phone number in E.164 format.
</ParamField>

<ParamField path="From" type="string" required={true} toc={true}>
  The caller ID phone number (must be a number you own).
</ParamField>

<ParamField path="Url" type="string" toc={true}>
  The URL that returns LAML instructions when the call connects.
</ParamField>

<ParamField path="StatusCallback" type="string" toc={true}>
  URL to receive call status webhook events.
</ParamField>

<ParamField path="StatusCallbackMethod" type="string" toc={true}>
  HTTP method for the `StatusCallback` URL. `"GET"` or `"POST"`.
</ParamField>

<ParamField path="StatusCallbackEvent" type="array" default='["completed"]' toc={true}>
  Events to send to the status callback.

  - `"initiated"` -- the call request has been created
  - `"ringing"` -- the destination is ringing
  - `"answered"` -- the call has been answered
  - `"completed"` -- the call has ended
</ParamField>

<ParamField path="ApplicationSid" type="string" toc={true}>
  The SID of the application to handle the call.
</ParamField>

<ParamField path="Method" type="string" toc={true}>
  HTTP method to use when requesting the `Url`. `"GET"` or `"POST"`.
</ParamField>

<ParamField path="FallbackUrl" type="string" toc={true}>
  A fallback URL if the primary `Url` fails.
</ParamField>

<ParamField path="FallbackMethod" type="string" toc={true}>
  HTTP method for the `FallbackUrl`. `"GET"` or `"POST"`.
</ParamField>

<ParamField path="CallerId" type="string" toc={true}>
  The caller ID to display for the call. Must be a verified number.
</ParamField>

<ParamField path="SendDigits" type="string" toc={true}>
  DTMF tones to play after the call connects. Use `w` for a half-second pause.
</ParamField>

<ParamField path="Timeout" type="number" toc={true}>
  Time in seconds to wait for the call to be answered before canceling.
</ParamField>

<ParamField path="MachineDetection" type="string" toc={true}>
  Enable answering machine detection.

  - `"Enable"` -- detect and return result in the callback
  - `"DetectMessageEnd"` -- detect and wait for the end of the machine's greeting
</ParamField>

<ParamField path="MachineDetectionTimeout" type="number" toc={true}>
  Time in milliseconds for the answering machine detection to complete.
</ParamField>

<ParamField path="MachineDetectionSpeechThreshold" type="number" toc={true}>
  Time in milliseconds of initial speech before classifying as human.
</ParamField>

<ParamField path="MachineDetectionSpeechEndThreshold" type="number" toc={true}>
  Silence time in milliseconds after speech that indicates speech has ended.
</ParamField>

<ParamField path="MachineDetectionSilenceTimeout" type="number" toc={true}>
  Time in milliseconds of initial silence before classifying as machine.
</ParamField>

<ParamField path="MachineWordsThreshold" type="number" toc={true}>
  Number of words in a greeting before classifying as machine.
</ParamField>

<ParamField path="AsyncAmd" type="boolean" toc={true}>
  Whether to perform answering machine detection asynchronously.
</ParamField>

<ParamField path="AsyncAmdStatusCallback" type="string" toc={true}>
  URL to receive async AMD results.
</ParamField>

<ParamField path="AsyncAmdStatusCallbackMethod" type="string" toc={true}>
  HTTP method for the `AsyncAmdStatusCallback`. `"GET"` or `"POST"`.
</ParamField>

<ParamField path="AsyncAmdPartialResults" type="boolean" toc={true}>
  Whether to send partial AMD results via callback.
</ParamField>

<ParamField path="Record" type="boolean" toc={true}>
  Whether to record the entire call.
</ParamField>

<ParamField path="RecordingChannels" type="string" toc={true}>
  Recording channel mode. `"mono"` or `"dual"`.
</ParamField>

<ParamField path="RecordingTrack" type="string" toc={true}>
  Which audio track to record. `"inbound"`, `"outbound"`, or `"both"`.
</ParamField>

<ParamField path="RecordingStatusCallback" type="string" toc={true}>
  URL to receive recording status events.
</ParamField>

<ParamField path="RecordingStatusCallbackMethod" type="string" toc={true}>
  HTTP method for the `RecordingStatusCallback`. `"GET"` or `"POST"`.
</ParamField>

<ParamField path="RecordingStatusCallbackEvent" type="string" toc={true}>
  Recording events to receive. `"in-progress"`, `"completed"`, `"absent"`.
</ParamField>

<ParamField path="Trim" type="string" toc={true}>
  Whether to trim silence from the recording. `"trim-silence"` or `"do-not-trim"`.
</ParamField>

<ParamField path="SipAuthUsername" type="string" toc={true}>
  SIP authentication username for SIP calls.
</ParamField>

<ParamField path="SipAuthPassword" type="string" toc={true}>
  SIP authentication password for SIP calls.
</ParamField>

<ParamField path="MaxPricePerMinute" type="string" toc={true}>
