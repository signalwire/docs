<ParamField path="play" type="array" required={true} toc={true}>
  Array of media objects to play. Each item has a `type` field and type-specific
  parameters. Supported types:

  - `"audio"` — Play an audio file. Requires `url`.
  - `"tts"` — Play text-to-speech. Requires `text`. Optional: `language`, `gender`, `voice`.
  - `"silence"` — Play silence. Requires `duration` (in seconds).
  - `"ring"` — Play a ring tone. Optional: `duration`.
</ParamField>

<ParamField path="volume" type="number" default="0" toc={true}>
  Volume adjustment in dB. Range: `-40` to `40`.
</ParamField>

<ParamField path="control_id" type="string" toc={true}>
  Unique identifier for this play operation, used to control it later.
  If not provided, one is generated automatically.
</ParamField>

<ParamField path="direction" type="string" default="both" toc={true}>
  The direction of audio playback.

  - `"listen"` — play to the listener only
  - `"speak"` — play to the speaker only
  - `"both"` — play to both legs
</ParamField>

<ParamField path="loop" type="integer" toc={true}>
  Number of times to loop playback. `0` for infinite loop.
</ParamField>

<ParamField path="status_url" type="string" toc={true}>
  A URL to receive status update callbacks for the play operation.
</ParamField>
