<ParamField path="action" type="object" required={true} toc={true}>
  The translation action to perform. Use one of:

  - `{ start: {...} }` — Start live translation. Accepts: `from_lang`, `to_lang`,
    `direction`, `from_voice`, `to_voice`, `filter_from`, `webhook`, `live_events`,
    `ai_summary`, `speech_engine`.
  - `{ summarize: {...} }` — Request a summary of the current translation session.
  - `{ inject: {...} }` — Inject text into the translation stream.
  - `"stop"` — Stop live translation.
</ParamField>

<ParamField path="status_url" type="string" toc={true}>
  A URL to receive status update callbacks for the translation session.
</ParamField>
