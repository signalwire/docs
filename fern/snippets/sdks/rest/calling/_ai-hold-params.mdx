<ParamField path="timeout" type="integer" default="300" toc={true}>
  The duration to hold the caller in seconds. Range: `1` to `300`.
</ParamField>

<ParamField path="prompt" type="string" default="Tell the user you are putting them on hold." toc={true}>
  A system message added to the AI conversation before placing the caller on hold.
  The AI will speak this message to the caller before hold music begins.
</ParamField>
