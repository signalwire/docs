<ParamField path="id" type="string" required={true} toc={true}>
  The unique identifying ID of the call to update.
</ParamField>

<ParamField path="url" type="string" toc={true}>
  URL to a new SWML document to control the call. Either `url` or `swml` is
  required.
</ParamField>

<ParamField path="swml" type="object" toc={true}>
  Inline SWML object containing new instructions for the call. Either `url` or
  `swml` is required.
</ParamField>

<ParamField path="fallback_url" type="string" toc={true}>
  Backup URL containing SWML instructions if the primary `url` fails.
</ParamField>

<ParamField path="status" type="string" toc={true}>
  Set the call status. Either `"canceled"` (cancel a not-yet-connected call) or
  `"completed"` (end a call in progress).
</ParamField>

<ParamField path="status_url" type="string" toc={true}>
  A URL to receive call status update callbacks.
</ParamField>
