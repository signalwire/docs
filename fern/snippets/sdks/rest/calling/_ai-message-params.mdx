<ParamField path="message_text" type="string" toc={true}>
  The text content that will be sent to the AI. Required when `reset` is not provided.
</ParamField>

<ParamField path="role" type="string" toc={true}>
  The role that the message is from. Required when `reset` is not provided.

  - `"system"` — Inject instructions or context that modify the AI's behavior mid-conversation.
  - `"user"` — Inject a message as if the caller said it.
  - `"assistant"` — Inject a message as if the AI said it.
</ParamField>

<ParamField path="reset" type="object" toc={true}>
  Parameters for resetting the AI conversation state. When provided, `role` and
  `message_text` are optional. Contains optional keys: `full_reset`, `user_prompt`,
  `system_prompt`.
</ParamField>

<ParamField path="global_data" type="object" toc={true}>
  Arbitrary JSON data to merge into the AI session's global data store.
</ParamField>
