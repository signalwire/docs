<ParamField path="reason" type="string" toc={true}>
  The reason for ending the call.

  - `"hangup"` -- normal call termination
  - `"busy"` -- end the call with a busy signal
  - `"cancel"` -- cancel the call before it is answered
  - `"noAnswer"` -- end the call as unanswered
  - `"decline"` -- decline the call
  - `"error"` -- end the call due to an error
</ParamField>
