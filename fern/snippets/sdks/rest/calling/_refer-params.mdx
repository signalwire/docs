<ParamField path="device" type="object" required={true} toc={true}>
  The SIP device to refer the call to. Contains:

  - `type` — Must be `"sip"`
  - `params` — Device parameters:
    - `to` — SIP URI to refer to (e.g., `"sip:user@example.com"`)
    - `username` — Optional SIP authentication username
    - `password` — Optional SIP authentication password
</ParamField>

<ParamField path="status_url" type="string" toc={true}>
  A URL to receive status update callbacks for the refer operation.
</ParamField>
