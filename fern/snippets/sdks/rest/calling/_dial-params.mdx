<ParamField path="from" type="string" required={true} toc={true}>
  The originating address. An E.164 phone number (`+xxxxxxxxxxx`) or SIP
  endpoint (`sip:xxx@yyy.zzz`).
</ParamField>

<ParamField path="to" type="string" required={true} toc={true}>
  The destination address. An E.164 phone number or SIP endpoint.
</ParamField>

<ParamField path="url" type="string" toc={true}>
  URL to a SWML document that controls the call. Either `url` or `swml` is
  required.
</ParamField>

<ParamField path="swml" type="object" toc={true}>
  Inline SWML object containing instructions for handling the call. Either
  `url` or `swml` is required.
</ParamField>

<ParamField path="caller_id" type="string" toc={true}>
  The caller ID number in E.164 format.
</ParamField>

<ParamField path="fallback_url" type="string" toc={true}>
  Backup URL containing SWML instructions if the primary `url` fails.
</ParamField>

<ParamField path="status_url" type="string" toc={true}>
  A URL to receive call status update callbacks for events listed in
  `status_events`.
</ParamField>

<ParamField path="status_events" type="array" default='["ended"]' toc={true}>
  Call events to monitor. Valid values: `"answered"`, `"queued"`, `"initiated"`,
  `"ringing"`, `"ending"`, `"ended"`.
</ParamField>

<ParamField path="url_method" type="string" default="POST" toc={true}>
  HTTP method to use when requesting the `url`.
</ParamField>
