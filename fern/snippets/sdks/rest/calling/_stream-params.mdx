<ParamField path="url" type="string" required={true} toc={true}>
  The WebSocket URL to stream audio to.
</ParamField>

<ParamField path="codec" type="string" toc={true}>
  The audio codec to use for streaming (e.g., `"PCMU"`, `"PCMA"`, `"OPUS"`).
</ParamField>

<ParamField path="track" type="string" toc={true}>
  The audio track to stream.

  - `"inbound_track"` — stream audio from the inbound leg
  - `"outbound_track"` — stream audio from the outbound leg
  - `"both_tracks"` — stream audio from both legs
</ParamField>

<ParamField path="authorization_bearer_token" type="string" toc={true}>
  Bearer token for WebSocket authentication.
</ParamField>

<ParamField path="custom_parameters" type="object" toc={true}>
  Custom parameters to include in the WebSocket connection.
</ParamField>

<ParamField path="control_id" type="string" toc={true}>
  Unique identifier for this stream operation, used to control it later.
  If not provided, one is generated automatically.
</ParamField>
