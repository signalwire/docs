<ParamField path="tap" type="object" required={true} toc={true}>
  Tap configuration specifying what audio to capture. Contains:

  - `type` — `"audio"` for audio tapping
  - `params` — Tap parameters including `direction` (`"listen"`, `"speak"`, or `"both"`)
</ParamField>

<ParamField path="device" type="object" required={true} toc={true}>
  Device configuration specifying where to stream captured audio. Contains:

  - `type` — `"rtp"` (RTP stream) or `"ws"` (WebSocket)
  - `params` — Connection parameters (e.g., `{"addr": "192.168.1.100", "port": 5060}` for RTP, or `{"uri": "wss://example.com/tap"}` for WebSocket)
</ParamField>

<ParamField path="control_id" type="string" toc={true}>
  Unique identifier for this tap operation, used to control it later.
  If not provided, one is generated automatically.
</ParamField>
