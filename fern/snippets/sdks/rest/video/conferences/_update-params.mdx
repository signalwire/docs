  Display name of the video conference. Maximum of 200 characters.

  Description of the conference. Maximum of 3000 characters.
</ParamField>

<ParamField path="join_from" type="string" toc={true}>
  Conference does not accept new participants before this time. Expects RFC 3339 datetime: `2022-01-01T23:59:60Z`. Date only: `2022-01-01` will be converted to `2022-01-01T00:00:00Z`.
</ParamField>

<ParamField path="join_until" type="string" toc={true}>
  Conference stops accepting new participants at this time, but keeps running until all participants leave. Expects RFC 3339 datetime.
</ParamField>

<ParamField path="quality" type="string" default="720p" toc={true}>
  The conference's resolution. One of `720p`, `1080p`.
</ParamField>

<ParamField path="layout" type="string" default="grid-responsive" toc={true}>

</ParamField>

<ParamField path="size" type="string" default="medium" toc={true}>

<ParamField path="tone_on_entry_and_exit" type="boolean" toc={true}>
