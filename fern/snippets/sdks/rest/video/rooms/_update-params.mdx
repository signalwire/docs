<ParamField path="display_name" type="string" toc={true}>
  Display name of the room. Maximum of 200 characters. Defaults to the value of `name`.
</ParamField>

<ParamField path="description" type="string" toc={true}>
  Description of the room. Maximum of 3000 characters.
</ParamField>

<ParamField path="max_members" type="integer" default="20" toc={true}>
  Maximum number of members in the room at a time. Must be between 1 and 300.
</ParamField>

<ParamField path="quality" type="string" default="720p" toc={true}>
  The room's resolution. One of `720p`, `1080p`.
</ParamField>

<ParamField path="join_from" type="string" toc={true}>
  Room does not accept new participants before this time. Expects RFC 3339 datetime: `2022-01-01T23:59:60Z`. Date only: `2022-01-01` will be converted to `2022-01-01T00:00:00Z`.
</ParamField>

<ParamField path="join_until" type="string" toc={true}>
  Room stops accepting new participants at this time, but keeps running until all participants leave. Expects RFC 3339 datetime.
</ParamField>

<ParamField path="remove_at" type="string" toc={true}>
  Remove users from the room at this time. Expects RFC 3339 datetime.
</ParamField>

<ParamField path="remove_after_seconds_elapsed" type="integer" toc={true}>
  Remove users after they are in the room for N seconds. Between 1 and 200000.
</ParamField>

<ParamField path="layout" type="string" default="grid-responsive" toc={true}>
  The room's initial layout. One of `grid-responsive`, `grid-responsive-mobile`, `highlight-1-responsive`, `1x1`, `2x1`, `2x2`, `5up`, `3x3`, `4x4`, `5x5`, `6x6`, `8x8`, `10x10`.
</ParamField>

<ParamField path="record_on_start" type="boolean" default="false" toc={true}>
  Whether to start recording a room session when one is started for this room.
</ParamField>

<ParamField path="enable_room_previews" type="boolean" default="false" toc={true}>
  Whether a video with a preview of the room content is generated.
</ParamField>

<ParamField path="meta" type="object" toc={true}>
  User-defined metadata for the room. Must be a valid JSON object. Maximum of 2000 characters when serialized.
</ParamField>

<ParamField path="sync_audio_video" type="boolean" toc={true}>
  Enable/disable jitter buffer audio-video sync.
</ParamField>
