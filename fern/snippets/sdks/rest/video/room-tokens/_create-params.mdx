<ParamField path="room_name" type="string" required={true} toc={true}>
  Room's unique named identifier. Allowed characters: `A-Za-z0-9_-`. Up to 100 characters. The room does not have to exist when the token is created, but must exist prior to joining (or set `auto_create_room` to `true`).
</ParamField>

<ParamField path="user_name" type="string" toc={true}>
  A display name for the user. Up to 100 characters. If not supplied, a random alphanumeric string is returned.
</ParamField>

<ParamField path="permissions" type="array" toc={true}>
  A list of permissions defining what the user can do once they join the room. If `join_as` is `"audience"`, permissions are set to an empty array regardless of the value provided.
</ParamField>

<ParamField path="join_from" type="string" toc={true}>
  The user cannot join the room before this time. Expects RFC 3339 datetime: `2022-01-01T23:59:60Z`. Date only: `2022-01-01` will be converted to `2022-01-01T00:00:00Z`.
</ParamField>

<ParamField path="join_until" type="string" toc={true}>
  The user cannot join the room after this time. Expects RFC 3339 datetime.
</ParamField>

<ParamField path="remove_at" type="string" toc={true}>
  Remove the user from the room at this time. Expects RFC 3339 datetime.
</ParamField>

<ParamField path="remove_after_seconds_elapsed" type="integer" toc={true}>
  Remove the user after they are in the room for N seconds. Maximum 200000.
</ParamField>

<ParamField path="join_audio_muted" type="boolean" default="false" toc={true}>
  Whether the user joins the room with their audio muted.
</ParamField>

<ParamField path="join_video_muted" type="boolean" default="false" toc={true}>
  Whether the user joins the room with their video muted.
</ParamField>

<ParamField path="auto_create_room" type="boolean" default="true" toc={true}>
  If the user tries to join a room that does not exist, it will be created with default configuration. Set to `false` to require the room to exist beforehand.
</ParamField>

<ParamField path="enable_room_previews" type="boolean" default="false" toc={true}>
  Whether to generate a video preview of the room content. Only takes effect if the token auto-creates the room.
</ParamField>

<ParamField path="room_display_name" type="string" toc={true}>
  Display name used if a room is auto-created when the token joins. Maximum of 200 characters. Defaults to the value of `room_name`.
</ParamField>

<ParamField path="end_room_session_on_leave" type="boolean" default="false" toc={true}>
  Whether to end the room session when the member using this token leaves the room.
</ParamField>

<ParamField path="join_as" type="string" default="member" toc={true}>
  Whether the user should join as a `"member"` or as a non-interactive `"audience"` participant. Audience participants cannot send audio or video.
</ParamField>

<ParamField path="media_allowed" type="string" default="all" toc={true}>
  Indicates what media the user is allowed to receive. One of `"all"`, `"audio-only"`, `"video-only"`.
</ParamField>

<ParamField path="room_meta" type="object" toc={true}>
  Set the room meta. Maximum of 2000 characters when serialized to JSON.
</ParamField>

<ParamField path="meta" type="object" toc={true}>
  Set the member meta. Maximum of 2000 characters when serialized to JSON.
</ParamField>

<ParamField path="sync_audio_video" type="boolean" default="false" toc={true}>
  Enable/disable jitter buffer audio-video sync.
</ParamField>
