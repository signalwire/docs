<ParamField path="query_string" type="string" required={true} toc={true}>
  The natural-language search query.
</ParamField>

<ParamField path="tags" type="array" toc={true}>
  Filter results to documents matching these tags. Array of strings.
</ParamField>

<ParamField path="document_id" type="string" toc={true}>
  Limit search to a specific document by its unique ID.
</ParamField>

<ParamField path="distance" type="number" toc={true}>
  Maximum distance threshold for results. Low distance means high relevance and similarity.
</ParamField>

<ParamField path="count" type="integer" default="5" toc={true}>
  Maximum number of result chunks to return.
</ParamField>

<ParamField path="language" type="string" default="en" toc={true}>
  Language of the document for query expansion.
</ParamField>

<ParamField path="pos_to_expand" type="array" toc={true}>
  Parts of speech considered for expansion or analysis. Array of strings (e.g., `NOUN`, `VERB`, `ADJ`, `ADV`). Defaults to `["NOUN", "VERB", "ADJ", "ADV"]`.
</ParamField>

<ParamField path="max_synonyms" type="integer" default="10" toc={true}>
  Maximum number of synonyms to consider for query expansion.
</ParamField>
