<ParamField path="url" type="string" required={true} toc={true}>
  URL of the document to ingest.
</ParamField>

<ParamField path="tags" type="array" toc={true}>
  Tags to associate with the document for filtering and retrieval. Array of strings.
</ParamField>

<ParamField path="chunking_strategy" type="string" toc={true}>
  Strategy for chunking the document. One of `sentence`, `sliding`, `page`, `paragraph`.
</ParamField>

<ParamField path="max_sentences_per_chunk" type="integer" default="50" toc={true}>
  Maximum number of sentences per chunk. Only applies when `chunking_strategy` is `sentence`.
</ParamField>

<ParamField path="split_newlines" type="boolean" default="false" toc={true}>
  Whether to split chunks on new lines. Only applies when `chunking_strategy` is `sentence`.
</ParamField>

<ParamField path="chunk_size" type="integer" default="50" toc={true}>
  Number of words per chunk. Only applies when `chunking_strategy` is `sliding`.
</ParamField>

<ParamField path="overlap_size" type="integer" default="10" toc={true}>
  Amount of overlap between chunks, in number of words. Only applies when `chunking_strategy` is `sliding`.
</ParamField>
