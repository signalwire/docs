Use this endpoint for the [Fax](/compatibility-api/index) method to list all faxes on your SignalWire account.

<EndpointSchemaSnippet endpoint="GET /Accounts/{AccountSid}/Faxes" />

<EndpointResponseSnippet endpoint="GET /Accounts/{AccountSid}/Faxes" />

## Request examples

<Tabs groupId="compatibility-sdks">

<Tab title="Node.js">

```javascript
const { RestClient } = require('@signalwire/compatibility-api')
const client = RestClient('YourProjectID', 'YourAuthToken', { signalwireSpaceUrl: 'example.signalwire.com' })

client.fax.faxes.each(faxes => console.log(faxes.sid));
```

</Tab>

<Tab title="C#">

```csharp
using System;
using System.Collections.Generic;
using Twilio;
using Twilio.Rest.Fax.V1;

class Program
{
    static void Main(string[] args)
    {
        TwilioClient.Init("YourProjectID", "YourAuthToken", new Dictionary<string, object> { ["signalwireSpaceUrl"] = "{SPACE}.signalwire.com" });

        var faxes = FaxResource.Read();

        foreach(var record in faxes)
        {
           Console.WriteLine(record.Sid);
        }
    }
}
```

</Tab>

<Tab title="Python">

```python
from signalwire.rest import Client as signalwire_client

client = signalwire_client("YourProjectID", "YourAuthToken", signalwire_space_url = 'example.signalwire.com')

faxes = client.fax.faxes.list()

for record in faxes:
    print(record.sid)
```
