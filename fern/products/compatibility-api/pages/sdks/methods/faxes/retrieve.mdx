Use this endpoint for the [Fax](/compatibility-api/index) method to retrieve a single fax by its  [SID](/platform/dashboard/guides/what-is-a-sid/).

<EndpointSchemaSnippet endpoint="GET /Accounts/{AccountSid}/Faxes/{Sid}" />

<EndpointResponseSnippet endpoint="GET /Accounts/{AccountSid}/Faxes/{Sid}" />

## Request examples

<Tabs groupId="compatibility-sdks">

<Tab title="Node.js">

```javascript
const { RestClient } = require('@signalwire/compatibility-api')
const client = RestClient('YourProjectID', 'YourAuthToken', { signalwireSpaceUrl: 'example.signalwire.com' })

client.fax.faxes('FaxSid')
          .fetch()
          .then(fax => console.log(fax.to))
          .done();
```

</Tab>

<Tab title="C#">

```csharp
using System;
using System.Collections.Generic;
using Twilio;
using Twilio.Rest.Fax.V1;

class Program
{
    static void Main(string[] args)
    {
        TwilioClient.Init("YourProjectID", "YourAuthToken", new Dictionary<string, object> { ["signalwireSpaceUrl"] = "{SPACE}.signalwire.com" });

        var fax = FaxResource.Fetch(pathSid: "FaxSid");

        Console.WriteLine(fax.To);
    }
}
```

</Tab>

<Tab title="Python">

```python
from signalwire.rest import Client as signalwire_client

client = signalwire_client("YourProjectID", "YourAuthToken", signalwire_space_url = 'example.signalwire.com')

fax = client.fax.faxes('FaxSid').fetch()

print(fax.to)
```
