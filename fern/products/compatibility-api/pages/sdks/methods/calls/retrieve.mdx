Use this endpoint for the [Calls](/compatibility-api/index) method to retrieve a single call.

<EndpointSchemaSnippet endpoint="GET /Accounts/{AccountSid}/Calls/{Sid}" />

<EndpointResponseSnippet endpoint="GET /Accounts/{AccountSid}/Calls/{Sid}" />

## Request examples

<Tabs groupId="compatibility-sdks">

<Tab title="Node.js">

```javascript
const { RestClient } = require('@signalwire/compatibility-api')
const client = RestClient('YourProjectID', 'YourAuthToken', { signalwireSpaceUrl: 'example.signalwire.com' })

client.calls('Sid')
      .fetch()
      .then(call => console.log(call.to))
      .done();
```

</Tab>

<Tab title="C#">

```csharp
using System;
using System.Collections.Generic;
using Twilio;
using Twilio.Rest.Api.V2010.Account;

class Program
{
    static void Main(string[] args)
    {
        TwilioClient.Init("YourProjectID", "YourAuthToken", new Dictionary<string, object> { ["signalwireSpaceUrl"] = "{SPACE}.signalwire.com" });

        var call = CallResource.Fetch(pathSid: "Sid");

        Console.WriteLine(call.To);
    }
}
```

</Tab>

<Tab title="Python">

```python
from signalwire.rest import Client as signalwire_client

client = signalwire_client("YourProjectID", "YourAuthToken", signalwire_space_url = 'example.signalwire.com')

call = client.calls('Sid').fetch()

print(call.to)
```

</Tab>
