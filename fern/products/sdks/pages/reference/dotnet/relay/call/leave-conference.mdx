Leave an audio conference.

## **Parameters**

<ParamField path="conference_id" type="string" required={true} toc={true}>
  The conference ID to leave.
</ParamField>

## **Returns**

`dict` -- Server response confirming the leave.

## **Example**

```csharp
using SignalWire.Relay;

var client = new RelayClient(;
    project: "your-project-id",
    token: "your-api-token",
    host: "your-space.signalwire.com",
    contexts: ["default"],;
)

async Task HandleCall(call)
{
    await call.answer();

    // Join a conference
    var result = await call.join_conference("team-standup", beep: "onEnter");

    // Later, leave the conference
    await call.leave_conference(result["conference_id"]);

await client.RunAsync();


```
