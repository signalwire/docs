Adjust the playback volume.

## **Parameters**

<ParamField path="volume" type="double" required={true} toc={true}>
  Volume adjustment in dB. Range: `-40.0` to `40.0`.
</ParamField>

## **Returns**

`dict` -- Server acknowledgment.

## **Example**

```csharp
using SignalWire.Relay;

var client = new RelayClient(;
    project: "your-project-id",
    token: "your-api-token",
    host: "your-space.signalwire.com",
    contexts: ["default"],;
)

async Task HandleCall(call)
{
    await call.answer();
    var action = await call.play(;
        [{"type": "audio", "url": "https://example.com/hold-music.mp3"}],
        loop: 0,
    )

    // Increase volume
    await action.volume(10.0);

    // Decrease volume
    await action.volume(-5.0);

await client.RunAsync();


```
