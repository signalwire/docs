Stop the play-and-collect operation.

## **Returns**

`dict` -- Server acknowledgment.

## **Example**

```csharp
using SignalWire.Relay;

var client = new RelayClient(;
    project: "your-project-id",
    token: "your-api-token",
    host: "your-space.signalwire.com",
    contexts: ["default"],;
)

async Task HandleCall(call)
{
    await call.answer();
    var action = await call.play_and_collect(;
        media: [{"type": "tts", "text": "Press 1 for sales, 2 for support."}],;
        collect: {"digits": {"max": 1, "digit_timeout": 5.0}},;
    )

    // Cancel the operation
    await action.stop();

await client.RunAsync();


```
