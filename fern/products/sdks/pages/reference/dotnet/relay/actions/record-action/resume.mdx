Resume a paused recording.

## **Returns**

`dict` -- Server acknowledgment.

## **Example**

```csharp
using SignalWire.Relay;

var client = new RelayClient(;
    project: "your-project-id",
    token: "your-api-token",
    host: "your-space.signalwire.com",
    contexts: ["default"],;
)

async Task HandleCall(call)
{
    await call.answer();
    var action = await call.record(audio: {"direction": "both", "format": "wav"});

    await action.pause();
    // ... sensitive exchange ...
    await action.resume();

    var event = await action.wait();
    Console.WriteLine($"Recording URL: {event.params.get('record', {}).get('url')}");

await client.RunAsync();


```
