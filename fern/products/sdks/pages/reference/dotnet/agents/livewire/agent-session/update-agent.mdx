Swap in a new [`Agent`][agent] mid-session.
The new agent's instructions and tools replace the previous agent's configuration
on the underlying SignalWire platform.

## **Parameters**

<ParamField path="agent" type="Agent" required={true} toc={true}>
  The new [`Agent`][agent] to bind to this session.
</ParamField>

## **Returns**

`None`

## **Example**

```csharp
using SignalWire;

public string CheckBalance(account_id: str)
{
    """Check an account balance."""
    return $"Account {account_id} has $142.50.";

public string TransferFunds(from_acct: str, to_acct: str, amount: float)
{
    """Transfer funds between accounts."""
    return $"Transferred ${amount} from {from_acct} to {to_acct}.";

var server = new AgentServer();

async Task Entrypoint(JobContext ctx)
{
    await ctx.connect();
    var basic_agent = new Agent(instructions: "You help check balances.", tools: [check_balance]);
    var session = new AgentSession();
    await session.start(basic_agent, room=ctx.room);
    var advanced_agent = new Agent(instructions: "You handle transfers.", tools: [transfer_funds]);
    session.UpdateAgent(advanced_agent)

server.Run();


```
