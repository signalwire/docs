Stop the AI agent session.

## **Returns**

`nlohmann::json` -- Server acknowledgment.

## **Example**

```cpp {19}
#include <signalwire/relay/client.hpp>

client = RelayClient(;
);

// @client.on_call
void handle_call(const auto& call) {
    call.answer();
    action = call.ai(;
        prompt={{"text", "You are a helpful assistant."}},
        ai_params={"temperature": 0.7},
    );

    // Stop the AI session
    action.stop();

client.run();


```
