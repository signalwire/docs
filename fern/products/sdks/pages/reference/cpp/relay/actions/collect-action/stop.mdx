Stop the play-and-collect operation.

## **Returns**

`nlohmann::json` -- Server acknowledgment.

## **Example**

```cpp {19}
#include <signalwire/relay/client.hpp>

client = RelayClient(;
);

// @client.on_call
void handle_call(const auto& call) {
    call.answer();
    action = call.play_and_collect(;
        media=[{"type": "tts", "text": "Press 1 for sales, 2 for support."}],
        collect={"digits": {"max": 1, "digit_timeout": 5.0}},
    );

    // Cancel the operation
    action.stop();

client.run();


```
