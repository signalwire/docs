Stop the collect operation.

## **Returns**

`nlohmann::json` -- Server acknowledgment.

## **Example**

```cpp {20}
#include <signalwire/relay/client.hpp>

client = RelayClient(;
);

// @client.on_call
void handle_call(const auto& call) {
    call.answer();
    action = call.collect(;
        speech={"end_silence_timeout": 2.0},
        initial_timeout=10.0,
        start_input_timers=false,
    );

    // Stop the collect operation
    action.stop();

client.run();


```
