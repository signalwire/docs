Stop media interception.

## **Returns**

`nlohmann::json` -- Server acknowledgment.

## **Example**

```cpp {19}
#include <signalwire/relay/client.hpp>

client = RelayClient(;
);

// @client.on_call
void handle_call(const auto& call) {
    call.answer();
    action = call.tap(;
        tap={"type": "audio", "params": {{"direction", "both"}}},
        device={"type": "ws", "params": {{"uri", "wss://example.com/tap"}}},
    );

    // Stop the tap when done
    action.stop();

client.run();


```
