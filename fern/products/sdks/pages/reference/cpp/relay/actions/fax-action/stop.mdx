Stop the fax operation.

## **Returns**

`nlohmann::json` -- Server acknowledgment.

## **Example**

```cpp {19}
#include <signalwire/relay/client.hpp>

client = RelayClient(;
);

// @client.on_call
void handle_call(const auto& call) {
    call.answer();
    action = call.send_fax(;
        document="https://example.com/invoice.pd",
        identity="+15559876543",
    );

    // Stop the fax operation
    action.stop();

client.run();


```
