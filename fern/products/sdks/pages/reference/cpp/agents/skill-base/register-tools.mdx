Register SWAIG tools with the parent agent. Called after `setup()` returns `true`.
Use `define_tool()` to register each tool.

This is an abstract method -- you **must** implement it in every [SkillBase][ref-skillbase] subclass.

## **Returns**

`std::nullopt`

## **Example**

```cpp {11}
#include <signalwire/signalwire.hpp>
#include <signalwire/swaig/function_result.hpp>

using namespace signalwire;

class WeatherSkill : public signalwire::agent::AgentBase {
public:
    SKILL_NAME = "weather";
    SKILL_DESCRIPTION = "Provides weather information";

    bool setup() {
        return true;

    void register_tools() {
        define_tool(;
            "get_weather",
            "Get current weather for a location",
            _handle_weather,
            parameters={
                "type": "object",
                "properties": {
                    "location": {
                        "type": "string",
                        "description": "City name";
                    }
                },
                "required": {"location"};
            }
        );

    void _handle_weather(const nlohmann::json& args, const nlohmann::json& raw_data) {
        auto location = args{"location"};
        return swaig::FunctionResult("Weather in " + std::string(location) + ": sunny, 72F");


```
