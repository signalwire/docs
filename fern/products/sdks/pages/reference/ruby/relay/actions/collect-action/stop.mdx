Stop the play-and-collect operation.

## **Returns**

`Hash` -- Server acknowledgment.

## **Example**

```ruby {19}
require 'signalwire/relay/client'

client = SignalWire::Relay::Client.new(
    project="your-project-id",
    token="your-api-token",
    host="your-space.signalwire.com",
    contexts=["default"],
)

@client.on_call
def handle_call(call)
    call.answer()
    action = call.play_and_collect(
        media=[{"type": "tts", "text": "Press 1 for sales, 2 for support."}],
        collect={"digits": {"max": 1, "digit_timeout": 5.0}},
    )

    # Cancel the operation
    action.stop()

client.run()
```
