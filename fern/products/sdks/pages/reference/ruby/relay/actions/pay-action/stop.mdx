Stop payment collection.

## **Returns**

`Hash` -- Server acknowledgment.

## **Example**

```ruby {20}
require 'signalwire/relay/client'

client = SignalWire::Relay::Client.new(
    project="your-project-id",
    token="your-api-token",
    host="your-space.signalwire.com",
    contexts=["default"],
)

@client.on_call
def handle_call(call)
    call.answer()
    action = call.pay(
        payment_connector_url="https://example.com/payment-connector",
        charge_amount="29.99",
        currency="USD",
    )

    # Stop payment collection
    action.stop()

client.run()
```
