Stop the collect operation.

## **Returns**

`Hash` -- Server acknowledgment.

## **Example**

```ruby {20}
require 'signalwire/relay/client'

client = SignalWire::Relay::Client.new(
    project="your-project-id",
    token="your-api-token",
    host="your-space.signalwire.com",
    contexts=["default"],
)

@client.on_call
def handle_call(call)
    call.answer()
    action = call.collect(
        speech={"end_silence_timeout": 2.0},
        initial_timeout=10.0,
        start_input_timers=false,
    )

    # Stop the collect operation
    action.stop()

client.run()
```
