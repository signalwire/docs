Resume a paused recording.

## **Returns**

`Hash` -- Server acknowledgment.

## **Example**

```ruby {17}
require 'signalwire/relay/client'

client = SignalWire::Relay::Client.new(
    project="your-project-id",
    token="your-api-token",
    host="your-space.signalwire.com",
    contexts=["default"],
)

@client.on_call
def handle_call(call)
    call.answer()
    action = call.record(audio: {"direction": "both", "format": "wav"})

    action.pause()
    # ... sensitive exchange ...
    action.resume()

    event = action.wait()
    puts("Recording URL: #{event.params.get('record', {})["url"]}")

client.run()
```
