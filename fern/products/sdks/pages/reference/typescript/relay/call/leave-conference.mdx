Leave an audio conference.

## **Parameters**

<ParamField path="conferenceId" type="string" required={true} toc={true}>
  The conference ID to leave.
</ParamField>

## **Returns**

`dict` -- Server response confirming the leave.

## **Example**

```typescript {18}
import { RelayClient } from "@signalwire/sdk";

client = RelayClient(
    project="your-project-id",
    token="your-api-token",
    host="your-space.signalwire.com",
    contexts=["default"],
)

@client.on_call
async def handle_call(call):
    await call.answer()

    # Join a conference
    result = await call.joinConference("team-standup", beep="onEnter")

    # Later, leave the conference
    await call.leaveConference(result["conference_id"])

client.run()
```
