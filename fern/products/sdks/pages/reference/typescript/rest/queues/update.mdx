Update a queue. Uses PUT.

## **Parameters**

<ParamField path="resourceId" type="string" required={true} toc={true}>
  The queue resource ID.
</ParamField>

<ParamField path="**kwargs" type="dict" toc={true}>
  Fields to update (e.g., `name`, `max_size`).
</ParamField>

## **Returns**

`dict` -- The updated queue object.

## **Example**

```typescript {9}
import { RestClient } from "@signalwire/sdk";

client = RestClient(
    project="your-project-id",
    token="your-api-token",
    host="your-space.signalwire.com",
)

updated = client.queues.update("queue-id", name="priority-support")
console.log(`Updated: ${updated['id']}`)
```
