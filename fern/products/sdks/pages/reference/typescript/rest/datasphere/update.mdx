Update an existing document.

## **Parameters**

<ParamField path="resourceId" type="string" required={true} toc={true}>
  The document resource ID.
</ParamField>

<ParamField path="**kwargs" type="dict" toc={true}>
  Fields to update (e.g., `name`, `content`, `metadata`).
</ParamField>

## **Returns**

`dict` -- The updated document object.

## **Example**

```typescript {9}
import { RestClient } from "@signalwire/sdk";

client = RestClient(
    project="your-project-id",
    token="your-api-token",
    host="your-space.signalwire.com",
)

updated = client.datasphere.documents.update(
    "document-id",
    name="Updated FAQ",
)
console.log(updated["name"])
```
