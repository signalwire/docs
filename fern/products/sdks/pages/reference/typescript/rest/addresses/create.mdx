Create a new regulatory address.

## **Parameters**

<ParamField path="label" type="string" required={true} toc={true}>
  A label for the address.
</ParamField>

<ParamField path="first_name" type="string" required={true} toc={true}>
  First name of the address holder.
</ParamField>

<ParamField path="last_name" type="string" required={true} toc={true}>
  Last name of the address holder.
</ParamField>

<ParamField path="street_number" type="string" required={true} toc={true}>
  Street number.
</ParamField>

<ParamField path="street_name" type="string" required={true} toc={true}>
  Street name.
</ParamField>

<ParamField path="city" type="string" required={true} toc={true}>
  City name.
</ParamField>

<ParamField path="state" type="string" required={true} toc={true}>
  State or province.
</ParamField>

<ParamField path="postal_code" type="string" required={true} toc={true}>
  Postal or ZIP code.
</ParamField>

<ParamField path="country" type="string" required={true} toc={true}>
  ISO country code (e.g., `"US"`).
</ParamField>

## **Returns**

`Promise<any>` -- The newly created address object.

## **Example**

```typescript {9}
import { RestClient } from "@signalwire/sdk";

const client = new RestClient({
  project: "your-project-id",
  token: "your-api-token",
  host: "your-space.signalwire.com"
});

const addr = await client.addresses.create({
    label: "HQ Address",
    first_name: "Jane",
    last_name: "Doe",
    street_number: "123",
    street_name: "Main St",
    city: "Austin",
    state: "TX",
    postal_code: "78701",
    country: "US",
});
console.log("Created address:", addr.id);
```
