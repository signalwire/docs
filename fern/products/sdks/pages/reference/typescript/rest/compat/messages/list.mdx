List messages in the account.

## **Parameters**

<ParamField path="To" type="string" toc={true}>
  Filter by destination number.
</ParamField>

<ParamField path="From" type="string" toc={true}>
  Filter by sender number.
</ParamField>

<ParamField path="DateSent" type="string" toc={true}>
  Filter by date sent (ISO 8601 format).
</ParamField>

<ParamField path="DateSent<" type="string" toc={true}>
  Filter by messages sent before this date (ISO 8601 format).
</ParamField>

<ParamField path="DateSent>" type="string" toc={true}>
  Filter by messages sent after this date (ISO 8601 format).
</ParamField>

<ParamField path="Status" type="string" toc={true}>
  Filter by message status.

  - `"queued"` -- message is queued for delivery
  - `"sending"` -- message is being sent
  - `"sent"` -- message has been sent
  - `"delivered"` -- message has been delivered
  - `"undelivered"` -- message could not be delivered
  - `"failed"` -- message delivery failed
  - `"received"` -- inbound message received
</ParamField>

<ParamField path="Page" type="number" default="0" toc={true}>
  Page number to retrieve.
</ParamField>

<ParamField path="PageSize" type="number" default="50" toc={true}>
  Number of results per page.
</ParamField>

<ParamField path="PageToken" type="string" toc={true}>
  Token for cursor-based pagination.
</ParamField>

## **Returns**

`Promise<any>` -- Paginated response containing message resources.

## **Example**

```typescript {8}
import { RestClient } from "@signalwire/sdk";

const client = new RestClient({
  project: "your-project-id",
  token: "your-api-token",
  host: "your-space.signalwire.com",
});

const messages = await client.compat.messages.list({ PageSize: 20 });
```
