List number groups in the project.

## **Parameters**

<ParamField path="**params" type="dict" toc={true}>
  Optional query parameters to filter and paginate results.
</ParamField>

## **Returns**

`dict` -- JSON response containing a `data` list of group objects.

## **Example**

```typescript {9}
import { RestClient } from "@signalwire/sdk";

client = RestClient(
    project="your-project-id",
    token="your-api-token",
    host="your-space.signalwire.com",
)

result = client.number_groups.list()
for group in result.get("data", []):
    console.log(group["name"], group["id"])
```
