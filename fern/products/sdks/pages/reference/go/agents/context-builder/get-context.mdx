Retrieve an existing context by name for inspection or modification.

## **Parameters**

<ParamField path="name" type="string" required={true} toc={true}>
  Name of the context to retrieve.
</ParamField>

## **Returns**

[`Context`][context] if found,
`nil` otherwise.

## **Example**

```go {9}
package main

import (
    "github.com/signalwire/signalwire-go/pkg/agent"
    "github.com/signalwire/signalwire-go/pkg/contexts"
)

func main() {
    agent := agent.NewAgentBase(agent.WithName("my-agent"), agent.WithRoute("/agent"))

    contexts := agent.define_contexts()
    default := contexts.add_context("default")
    default.AddStep("greet").set_text("Greet the caller.")

    ctx := contexts.get_context("default")
    if ctx is not nil:
        ctx.AddStep("followup").set_text("Ask if there is anything else.")

    agent.Serve()
}
```
