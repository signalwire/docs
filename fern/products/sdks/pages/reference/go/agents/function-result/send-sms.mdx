Send an SMS or MMS message to a phone number. At least one of `body` or `media`
must be provided. Generates a SWML `send_sms` verb under the hood.

## **Parameters**

<ParamField path="to_number" type="string" required={true} toc={true}>
  Destination phone number in E.164 format (e.g., `"+15551234567"`).
</ParamField>

<ParamField path="from_number" type="string" required={true} toc={true}>
  Sender phone number in E.164 format. Must be a number in your SignalWire project.
</ParamField>

<ParamField path="body" type="*string" default="None" toc={true}>
  Text body of the message.
</ParamField>

<ParamField path="media" type="*[]string" default="None" toc={true}>
  List of media URLs to include as MMS attachments.
</ParamField>

<ParamField path="tags" type="*[]string" default="None" toc={true}>
  Tags to associate with the message for searching and filtering in the
  SignalWire dashboard.
</ParamField>

<ParamField path="region" type="*string" default="None" toc={true}>
  Region to originate the message from.
</ParamField>

## **Returns**

[`FunctionResult`][functionresult] — self, for chaining.

## **Examples**

### Text Message

```go {13}
package main

import (
    "github.com/signalwire/signalwire-go/pkg/agent"
    "github.com/signalwire/signalwire-go/pkg/swaig"
)

func main() {
    agent := agent.NewAgentBase(agent.WithName("my-agent"), agent.WithRoute("/agent"))
    agent.SetPromptText("You are a helpful assistant.")

    agent.DefineTool(agent.ToolDefinition{
        Name:        "send_confirmation",
        Description: "Send an order confirmation SMS",
        Handler: func(args map[string]any, rawData map[string]any) *swaig.FunctionResult {
        phone := args[]string{"phone_number"}
        order_id := args[]string{"order_id"}
        return (
            FunctionResult("I've sent you a confirmation text.")
            .send_sms(
                to_number := phone,
                from_number := "+15559876543",
                body := fmt.Sprintf("Your order %v has been confirmed!", order_id)
            )
        )

    agent.Serve()
}
```

### MMS with Media

```go {13}
package main

import (
    "github.com/signalwire/signalwire-go/pkg/agent"
    "github.com/signalwire/signalwire-go/pkg/swaig"
)

func main() {
    agent := agent.NewAgentBase(agent.WithName("my-agent"), agent.WithRoute("/agent"))
    agent.SetPromptText("You are a helpful assistant.")

    agent.DefineTool(agent.ToolDefinition{
        Name:        "send_receipt",
        Description: "Send a receipt with media attachment",
        Handler: func(args map[string]any, rawData map[string]any) *swaig.FunctionResult {
        phone := args[]string{"phone_number"}
        receipt_url := args[]string{"receipt_url"}
        return (
            FunctionResult("I've sent your receipt.")
            .send_sms(
                to_number := phone,
                from_number := "+15559876543",
                body := "Here's your receipt:",
                media := [receipt_url]
            )
        )

    agent.Serve()
}
```
