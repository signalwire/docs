Remove one or more keys from the global session data.

## **Parameters**

<ParamField path="keys" type="interface{}" required={true} toc={true}>
  A single key string or a list of key strings to remove from global data.
</ParamField>

## **Returns**

[`FunctionResult`][functionresult] — self, for chaining.

## **Example**

```go {11}
package main

import (
    "github.com/signalwire/signalwire-go/pkg/agent"
    "github.com/signalwire/signalwire-go/pkg/swaig"
)

func main() {
    agent := agent.NewAgentBase(agent.WithName("my-agent"), agent.WithRoute("/agent"))
    agent.SetPromptText("You are a helpful assistant.")

    agent.DefineTool(agent.ToolDefinition{
        Name:        "clear_session_data",
        Description: "Clear session data",
        Handler: func(args map[string]any, rawData map[string]any) *swaig.FunctionResult {
        return (
            FunctionResult("Session data cleared.")
            .remove_global_data([]string{"customer_id", "verified"})
        )

    agent.Serve()
}
```
