Stop detection immediately.

## **Returns**

`dict` -- Server acknowledgment.

## **Example**

```go {18}
package main

import "github.com/signalwire/signalwire-go/pkg/relay"

func main() {
    client := RelayClient(
        project := "your-project-id",
        token := "your-api-token",
        host := "your-space.signalwire.com",
        contexts := []string{"default"},
    )

    async def handle_call(call):
        await call.answer()
        action := await call.detect(
            {"type": "machine", "params": {"initial_timeout": 5.0}}
        )

        // Stop detection early
        await action.stop()

    client.Run()
}
```
