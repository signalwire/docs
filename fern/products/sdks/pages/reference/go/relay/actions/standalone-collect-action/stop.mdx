Stop the collect operation.

## **Returns**

`dict` -- Server acknowledgment.

## **Example**

```go {20}
package main

import "github.com/signalwire/signalwire-go/pkg/relay"

func main() {
    client := RelayClient(
        project := "your-project-id",
        token := "your-api-token",
        host := "your-space.signalwire.com",
        contexts := []string{"default"},
    )

    async def handle_call(call):
        await call.answer()
        action := await call.collect(
            speech := {"end_silence_timeout": 2.0},
            initial_timeout := 10.0,
            start_input_timers := false,
        )

        // Stop the collect operation
        await action.stop()

    client.Run()
}
```
