Resume a paused recording.

## **Returns**

`dict` -- Server acknowledgment.

## **Example**

```go {17}
package main

import (
    "fmt"
    "github.com/signalwire/signalwire-go/pkg/relay"
)

func main() {
    client := RelayClient(
        project := "your-project-id",
        token := "your-api-token",
        host := "your-space.signalwire.com",
        contexts := []string{"default"},
    )

    async def handle_call(call):
        await call.answer()
        action := await call.record(audio={"direction": "both", "format": "wav"})

        await action.pause()
        // ... sensitive exchange ...
        await action.resume()

        event := await action.wait()
        fmt.Println(fmt.Sprintf("Recording URL: %v).get('url')}", event.params.get('record', {))

    client.Run()
}
```
