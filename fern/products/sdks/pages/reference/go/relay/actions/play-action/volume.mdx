Adjust the playback volume.

## **Parameters**

<ParamField path="volume" type="float64" required={true} toc={true}>
  Volume adjustment in dB. Range: `-40.0` to `40.0`.
</ParamField>

## **Returns**

`dict` -- Server acknowledgment.

## **Example**

```go {19,22}
package main

import "github.com/signalwire/signalwire-go/pkg/relay"

func main() {
    client := RelayClient(
        project := "your-project-id",
        token := "your-api-token",
        host := "your-space.signalwire.com",
        contexts := []string{"default"},
    )

    async def handle_call(call):
        await call.answer()
        action := await call.play(
            [{"type": "audio", "url": "https://example.com/hold-music.mp3"}],
            loop := 0,
        )

        // Increase volume
        await action.volume(10.0)

        // Decrease volume
        await action.volume(-5.0)

    client.Run()
}
```
