Stop media interception.

## **Returns**

`dict` -- Server acknowledgment.

## **Example**

```go {19}
package main

import "github.com/signalwire/signalwire-go/pkg/relay"

func main() {
    client := RelayClient(
        project := "your-project-id",
        token := "your-api-token",
        host := "your-space.signalwire.com",
        contexts := []string{"default"},
    )

    async def handle_call(call):
        await call.answer()
        action := await call.tap(
            tap := {"type": "audio", "params": {"direction": "both"}},
            device := {"type": "ws", "params": {"uri": "wss://example.com/tap"}},
        )

        // Stop the tap when done
        await action.stop()

    client.Run()
}
```
