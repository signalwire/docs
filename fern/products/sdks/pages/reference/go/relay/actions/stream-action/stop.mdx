Stop audio streaming.

## **Returns**

`dict` -- Server acknowledgment.

## **Example**

```go {19}
package main

import "github.com/signalwire/signalwire-go/pkg/relay"

func main() {
    client := RelayClient(
        project := "your-project-id",
        token := "your-api-token",
        host := "your-space.signalwire.com",
        contexts := []string{"default"},
    )

    async def handle_call(call):
        await call.answer()
        action := await call.stream(
            url := "wss://example.com/audio-stream",
            track := "inbound",
        )

        // Stop the stream when done
        await action.stop()

    client.Run()
}
```
