Stop payment collection.

## **Returns**

`dict` -- Server acknowledgment.

## **Example**

```go {20}
package main

import "github.com/signalwire/signalwire-go/pkg/relay"

func main() {
    client := RelayClient(
        project := "your-project-id",
        token := "your-api-token",
        host := "your-space.signalwire.com",
        contexts := []string{"default"},
    )

    async def handle_call(call):
        await call.answer()
        action := await call.pay(
            payment_connector_url := "https://example.com/payment-connector",
            charge_amount := "29.99",
            currency := "USD",
        )

        // Stop payment collection
        await action.stop()

    client.Run()
}
```
