Update the agent's system prompt mid-session. The new instructions take effect
immediately for subsequent LLM turns.

## **Parameters**

<ParamField path="instructions" type="string" required={true} toc={true}>
  The new system prompt text.
</ParamField>

## **Returns**

`null`

## **Example**

```php {18}
<?php
use SignalWire\Livewire\Agent;
use SignalWire\Livewire\AgentSession;
use SignalWire\Livewire\AgentServer;
use SignalWire\Livewire\JobContext;
use SignalWire\Livewire\function_tool;
use SignalWire\Livewire\run_app;

// @function_tool
function escalate(): string
{
    """Escalate to a senior agent."""
    return "Escalating now.";
}

$server = new AgentServer();

// @server.rtc_session()
function entrypoint(JobContext $ctx)
{
    $ctx->connect();
    $agent = $Agent(
        instructions: "You are a junior support agent. Help with basic questions.",
        tools: [$escalate],
    );
    $session = $AgentSession();
    $agent->update_instructions("You are now a senior support agent. Handle complex issues.");
    $session->start($agent, room: ctx.$room);
}

$run_app($server);


```
