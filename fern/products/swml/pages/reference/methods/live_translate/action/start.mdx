Start a live translation session.

<Schema type="StartAction" />

## **Example**

<CodeBlocks>
<CodeBlock title="YAML">
```yaml
live_translate:
  action:
    start:
      webhook: 'https://example.com/webhook'
      from_lang: en-US
      to_lang: es-ES
      from_voice: elevenlabs.josh
      to_voice: elevenlabs.josh
      filter_from: professional
      live_events: true
      ai_summary: true
      ai_summary_prompt: Summarize this conversation
      speech_timeout: 60000
      vad_silence_ms: 500
      vad_thresh: 400
      debug_level: 0
      direction:
        - remote-caller
        - local-caller
      speech_engine: deepgram
```
</CodeBlock>
