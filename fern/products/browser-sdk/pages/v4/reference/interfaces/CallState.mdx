## **Extended by**

- [`Call`](/docs/browser-sdk/v4/reference/interfaces/call)

## **Properties**

<ParamField path="id" type="string" required toc={true}>

</ParamField>

<ParamField path="locked" type="boolean" required toc={true}>
  Whether the call is locked.
</ParamField>

<ParamField path="locked$" type="Observable<boolean>" required toc={true}>

</ParamField>

<ParamField path="meta" type="Record<string, unknown>" required toc={true}>

</ParamField>

<ParamField path="meta$" type="Observable<Record<string, unknown>>" required toc={true}>

</ParamField>

<ParamField path="participants$" type="Observable<CallParticipant[]>" required toc={true}>
  Observable of the current participant list.
</ParamField>

<ParamField path="raiseHandPriority" type="boolean" required toc={true}>
  Whether raise-hand priority mode is active.
</ParamField>

<ParamField path="raiseHandPriority$" type="Observable<boolean>" required toc={true}>
  Observable of the raise-hand priority mode state.
</ParamField>

<ParamField path="recording" type="boolean" required toc={true}>
  Whether a server-side recording is active.
</ParamField>

<ParamField path="recording$" type="Observable<boolean>" required toc={true}>
  Observable of the current recording state.
</ParamField>

<ParamField path="status" type="CallStatus" required toc={true}>
  Current lifecycle status of the call.
</ParamField>

<ParamField path="status$" type="Observable<CallStatus>" required toc={true}>
  Observable of the call's lifecycle status.
</ParamField>

<ParamField path="streaming" type="boolean" required toc={true}>
  Whether a server-side stream is active.
</ParamField>

<ParamField path="streaming$" type="Observable<boolean>" required toc={true}>
  Observable of the current streaming state.
</ParamField>
