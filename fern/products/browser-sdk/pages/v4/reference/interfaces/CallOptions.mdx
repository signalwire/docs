Configuration options for creating a call.

## **Extends**

- [`MediaOptions`](MediaOptions.mdx)

## **Properties**

<ParamField path="callId" type="string" toc={true}>
  Pre-assigned call ID (used for reattach).
</ParamField>

<ParamField path="displayDirection" type="string" toc={true}>
  Direction hint for display purposes.
</ParamField>

<ParamField path="from" type="string" toc={true}>
  Address URI of the caller.
</ParamField>

<ParamField path="fromName" type="string" toc={true}>
  Display name of the caller.
</ParamField>

<ParamField path="initOffer" type="string" toc={true}>
  SDP offer for inbound calls.
</ParamField>

<ParamField path="nodeId" type="string" toc={true}>
  Optional. Hint to the cluster about which node should host this call. Used by reattach to pin to the original node; on fresh dials acts as a steering preference (the server may ignore for placement reasons). Leave undefined for normal load-balanced placement.
</ParamField>

<ParamField path="preferredAudioCodecs" type="string[]" toc={true}>
  Preferred audio codecs for this call (overrides global preferences).
</ParamField>

<ParamField path="preferredVideoCodecs" type="string[]" toc={true}>
  Preferred video codecs for this call (overrides global preferences).
</ParamField>

<ParamField path="reattach" type="boolean" toc={true}>
  Whether this call is being reattached after reconnect.
</ParamField>

<ParamField path="stereo" type="boolean" toc={true}>
  Enable stereo Opus for this call (overrides global preferences).
</ParamField>

<ParamField path="to" type="string" toc={true}>
  Destination URI.
</ParamField>

<ParamField path="toName" type="string" toc={true}>
  Display name of the callee.
</ParamField>

<ParamField path="userVariables" type="Record<string, unknown>" toc={true}>
  Custom user variables sent with the call invite.
</ParamField>

<Accordion title="Inherited from MediaOptions">
  <ParamField path="audio" type="boolean" toc={true}>
    Enable audio input. Defaults to `true` when not specified.
  </ParamField>

  <ParamField path="inputAudioDeviceConstraints" type="MediaTrackConstraints" toc={true}>
    Custom constraints for the audio input track.
  </ParamField>

  <ParamField path="inputAudioStream" type="MediaStream" toc={true}>
    Pre-existing audio stream to use instead of `getUserMedia`.
  </ParamField>

  <ParamField path="inputVideoDeviceConstraints" type="MediaTrackConstraints" toc={true}>
    Custom constraints for the video input track.
  </ParamField>

  <ParamField path="inputVideoStream" type="MediaStream" toc={true}>
    Pre-existing video stream to use instead of `getUserMedia`.
  </ParamField>

  <ParamField path="receiveAudio" type="boolean" toc={true}>
    Whether to receive remote audio.
  </ParamField>

  <ParamField path="receiveVideo" type="boolean" toc={true}>
    Whether to receive remote video.
  </ParamField>

  <ParamField path="video" type="boolean" toc={true}>
    Enable video input. Defaults to `false` when not specified.
  </ParamField>
</Accordion>
