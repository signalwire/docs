## Parameters

<ParamField path="chunking_strategy" type="str" default="'sentence'">
  Strategy for chunking documents:
- 'sentence': Sentence-based chunking with overlap
- 'sliding': Sliding window with word-based chunks
- 'paragraph': Natural paragraph boundaries
- 'page': Page-based chunking (for PDFs)
- 'semantic': Semantic similarity-based chunking
- 'topic': Topic modeling-based chunking
- 'qa': Question-answer optimized chunking
- 'json': JSON structure-aware chunking
- 'markdown': Markdown structure-aware chunking with code block detection
</ParamField>
<ParamField path="max_sentences_per_chunk" type="int" default="5">
  For sentence strategy (default: 5)
</ParamField>
<ParamField path="chunk_size" type="int" default="50">
  For sliding strategy - words per chunk (default: 50)
</ParamField>
<ParamField path="chunk_overlap" type="int" default="10">
  For sliding strategy - overlap in words (default: 10)
</ParamField>
<ParamField path="split_newlines" type="Optional[int]" default="None">
  For sentence strategy - split on multiple newlines (optional)
</ParamField>
<ParamField path="index_nlp_backend" type="str" default="'nltk'">
  NLP backend for indexing (default: 'nltk')
</ParamField>
<ParamField path="verbose" type="bool" default="False">
  Whether to enable verbose logging (default: False)
</ParamField>
<ParamField path="semantic_threshold" type="float" default="0.5">
  Similarity threshold for semantic chunking (default: 0.5)
</ParamField>
<ParamField path="topic_threshold" type="float" default="0.3">
  Similarity threshold for topic chunking (default: 0.3)
</ParamField>
