The REST namespace provides an HTTP client for the SignalWire platform APIs. It
organizes every HTTP endpoint into namespaced resource objects with standard
CRUD operations (`List`, `Create`, `Get`, `Update`, `Delete`), letting you
manage phone numbers, fabric resources, call logs, video rooms, datasphere
documents, and more from Go.

The constructor `rest.NewRestClient(project, token, space)` returns a
`*rest.RestClient` and a non-nil error when any of the three credentials is
empty.

## Example

Search for available phone numbers, purchase one, and list AI agent fabric
resources:

```go {10}
package main

import (
	"fmt"

	"github.com/signalwire/signalwire-go/pkg/rest"
)

func main() {
	client, err := rest.NewRestClient("your-project-id", "your-api-token", "your-space.signalwire.com")
	if err != nil {
		panic(err)
	}

	// Search for available numbers in area code 512
	available, err := client.PhoneNumbers.Search(map[string]any{
		"area_code": "512",
		"quantity":  3,
	})
	if err != nil {
		panic(err)
	}
	if data, ok := available["data"].([]any); ok {
		for _, n := range data {
			number, _ := n.(map[string]any)
			fmt.Printf("%v - %v\n", number["number"], number["region"])
		}
	}

	// List your AI agent resources
	agents, err := client.Fabric.AIAgents.List(nil)
	if err != nil {
		panic(err)
	}
	if data, ok := agents["data"].([]any); ok {
		for _, a := range data {
			agent, _ := a.(map[string]any)
			fmt.Printf("Agent: %v (%v)\n", agent["name"], agent["id"])
		}
	}

	// Query recent voice call logs
	logs, err := client.Logs.Voice.List(map[string]string{"page_size": "5"})
	if err != nil {
		panic(err)
	}
	if data, ok := logs["data"].([]any); ok {
		for _, l := range data {
			log, _ := l.(map[string]any)
			fmt.Printf("Call from %v to %v\n", log["from"], log["to"])
		}
	}
}
```

## Error Handling

REST errors are returned as `*rest.SignalWireRestError`:

```go {14-15}
package main

import (
	"errors"
	"fmt"

	"github.com/signalwire/signalwire-go/pkg/rest"
)

func main() {
	client, _ := rest.NewRestClient("project", "token", "space.signalwire.com")
	_, err := client.PhoneNumbers.Get("nonexistent-id")
	if err != nil {
		var rerr *rest.SignalWireRestError
		if errors.As(err, &rerr) {
			fmt.Printf("HTTP %d: %s\n", rerr.StatusCode, rerr.Body)
			fmt.Printf("URL: %s %s\n", rerr.Method, rerr.URL)
		}
	}
}
```
